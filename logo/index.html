Starburst

TO starburst :sides :length
  SETWIDTH 2
  SETCOLOR "orange"
  REPEAT :sides [
    FD :length
    BK :length
    RT 360 / :sides
  ]
END

starburst 36 150

Circle

# Draws a circle from the center
TO circle :radius
  PU
  HOME
  FD :radius
  LT 90
  PD
  
  MAKE "circumference 2 * 3.14159 * :radius
  MAKE "side :circumference / 360
  
  REPEAT 360 [
    FD :side
    LT 1
  ]
  
  PU
  HOME
  PD
END

SETCOLOR "red"
SETWIDTH 3
circle 150

3D Cube

TO square :size
